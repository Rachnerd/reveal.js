Decentralize the schema

ItemMS

type Item @key(fields: "id") {
    id: ID!
    name: String!
    description: String!
    image: String!
}

ReviewMS

type Review @key(fields: "userId") {
    id: ID!
    userId: ID!
    description: String!
    rating: Float!
}

extend type Item @key(fields: "id"){
    id: ID! @external
    averageRating: Float!
    reviews: [Review!]!
}

UserMS

type User @key(fields: "id") {
    id: ID!
    name: String!
    image: String!
}

extend type Item @key(fields: "userId") {
    userId: ID! @external
    seller: User!
}

extend type Review @key(fields: "userId") {
    userId: ID! @external
    author: User!
}
